using System;

public class 구구단
{
  public static void Main()
  {
    for (let i = 1; i <= 9; i++)
    {
      console.log(
        "{0} * {1} = {2}", 2, i, (2 * i));  
    }

    for (let i = 2; i <= 9; i++)
    {
      for (let j = 1; j <= 9; j++)
      {
        console.log(
          "{0} * {1} = {2}", i, j, (i * j));
      }  
    }

  }
}
